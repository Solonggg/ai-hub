// 剪贴板类 app 的"自动粘贴"时序
// 关自己 → 激活之前的 app → 模拟 ⌘V
// 50ms / 60ms 两段延时是实测稳定值；缩短会出现粘贴打到自己的现象。
import Cocoa

/// 在面板上记录当前 frontmost app，作为后续 activate 目标
final class PreviousAppTracker {
    private(set) var previousApp: NSRunningApplication?

    func record() {
        previousApp = NSWorkspace.shared.frontmostApplication
    }
}

/// 自动粘贴：close self → activate previous → simulate paste
func activatePreviousAppAndPaste(
    previousApp: NSRunningApplication?,
    closeSelf: () -> Void,
    simulatePaste: @escaping () -> Void
) {
    closeSelf()
    // 第一段：50ms 让自己的窗口关闭循环走完
    DispatchQueue.main.asyncAfter(deadline: .now() + 0.05) {
        if let app = previousApp, !app.isActive {
            app.activate(options: .activateIgnoringOtherApps)
        }
        // 第二段：60ms 让目标 app 完成激活并 key
        DispatchQueue.main.asyncAfter(deadline: .now() + 0.06) {
            simulatePaste()
        }
    }
}
