// 在 SwiftUI 里用 NSEvent.addLocalMonitor 监听键盘时，
// 中文/日文/韩文输入法组字期间候选窗口需要方向键切候选。
// 如果不放行，用户的输入法体验会被应用快捷键吃掉。
//
// 规则：hasMarkedText == true 直接 return event，让系统先处理。
import SwiftUI
import AppKit

final class KeyMonitorController {
    private var keyMonitor: Any?

    /// handler 返回 nil 表示消费事件，返回 event 表示放行
    func install(handler: @escaping (NSEvent) -> NSEvent?) {
        keyMonitor = NSEvent.addLocalMonitorForEvents(matching: .keyDown) { event in
            // 1. IME 组字期间一律放行，不要动
            if NSTextInputContext.current?.client.hasMarkedText() == true {
                return event
            }
            return handler(event)
        }
    }

    func remove() {
        if let m = keyMonitor {
            NSEvent.removeMonitor(m)
            keyMonitor = nil
        }
    }
}
